# Create simulator instance
set ns [new Simulator]

# Create nodes
set n0 [$ns node]
set n1 [$ns node]

# Label nodes
$ns at 0.0 "$n0 label Sender"
$ns at 0.0 "$n1 label Receiver"

# Open NAM and trace files
set nf [open A1-stop-n-wait.nam w]
$ns namtrace-all $nf
set tf [open A1-stop-n-wait.tr w]
$ns trace-all $tf

# Create duplex link with bandwidth, delay, and queue
$ns duplex-link $n0 $n1 0.2Mb 200ms DropTail
$ns duplex-link-op $n0 $n1 orient right
$ns queue-limit $n0 $n1 10 

# Set NAM trace variable for TCP
Agent/TCP set nam_tracevar_ truc

# Create TCP agent with Stop & Wait parameters
set tcp [new Agent/TCP]
$tcp set window_ 1
$tcp set maxcwnd_ 1
$ns attach-agent $n0 $tcp

# Create TCP sink and attach to receiver
set sink [new Agent/TCPSink]
$ns attach-agent $n1 $sink

# Connect TCP agent and sink
$ns connect $tcp $sink

# Create FTP application and attach to TCP agent
set ftp [new Application/FTP]
$ftp attach-agent $tcp

# Enable agent tracing
$ns add-agent-trace $tcp $tcp
$ns monitor-agent-trace $tcp
$tcp tracevar cwnd_

# Schedule start and stop times for FTP
$ns at 0.1 "$ftp start"
$ns at 3.0 "$ns detach-agent $n0 $tcp ; $ns detach-agent $n1 $sink"
$ns at 3.5 "finish"

# Trace annotations for NAM
$ns at 0.0 "$ns trace-annotate \"Stop and Wait with normal operation\""
$ns at 0.05 "$ns trace-annotate \"FTP starts at 0.1\""
$ns at 0.11 "$ns trace-annotate \"Send Packet_0\""
$ns at 0.35 "$ns trace-annotate \"Receive Ack_0\""
$ns at 0.56 "$ns trace-annotate \"Send Packet_1\""
$ns at 0.79 "$ns trace-annotate \"Receive Ack_1\""
$ns at 0.99 "$ns trace-annotate \"Send Packet_2\""
$ns at 1.23 "$ns trace-annotate \"Receive Ack_2\""

# Run simulation
$ns run

/*
Execution Steps:

1. Save this script in a file named stop_and_wait.tcl
   Command:
   $ nano stop_and_wait.tcl    (or any editor)

2. Make sure NS2 is installed on your system. To check:
   $ ns -version

3. Run the simulation:
   $ ns stop_and_wait.tcl

4. This will generate:
   - A1-stop-n-wait.nam (NAM trace file)
   - A1-stop-n-wait.tr (Trace file)
   - Console output showing Stop & Wait operation

5. To visualize the network animation:
   $ nam A1-stop-n-wait.nam

6. Observe:
   - Packet-by-packet transmission (window size = 1)
   - ACK reception after each packet
   - Congestion control effect (Stop & Wait)
*/
